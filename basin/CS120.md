##### CLI-Based

```run-java
import java.util.Scanner;

public class MyClass {
  public static void main(String args[]) {
    // Define Variables
    int numberOfCrates;
    int avgNumberApplesPerCrate;
    int expectedAmount;
    
    // Initialize variables using user input
    Scanner scnr = new Scanner(System.in);
    System.out.print("Number of crates to be received: ");
    numberOfCrates = scnr.nextInt();
    System.out.print("Average amount of apples per crate: ");
    avgNumberApplesPerCrate = scnr.nextInt();
    
    // Calculate average amount of apples
    expectedAmount = numberOfCrates * avgNumberApplesPerCrate;
    
    // Print the expectedAmount of apples in the latest shipment
    System.out.println("Expected number of apples in this shipment: " + expectedAmount);
  }
}
```

##### Using swing package with JOptionPane

```run-java
import javax.swing.JOptionPane;

public class MyClass {
  public static void main(String args[]) {
	// Initializing the userInput to prevent an error
	int firstUserInput = -1;
	int secondUserInput = -1;
	int expectedAmount = -1;
	// Create boolean toggles
	boolean firstInputValid = false;
	boolean secondInputValid = false;
	
  
	// Using a while loop to prompt the user continuously if they do not pass the right type of input or close the dialog box
	while (!firstInputValid) {
	
		// Create a new dialog box expecting user input, set to null by default
		String numberOfCrates = JOptionPane.showInputDialog(null, "How many crates do you expect to receive?");
		
		if (numberOfCrates == null) {
			JOptionPane.showMessageDialog(null, "Cancelled.");
			// Terminates the execution and passes control back to enclosing loop for the dialog box
			continue;
		}
		
		// Using the try keyword to test for exception errors as the user attempts to pass in relevant input
		try {
			firstUserInput = Integer.parseInt(numberOfCrates);
			// Flip the initial input into a true boolean value
			firstInputValid = true;
		// This part handles the caught error within the try code block that is testing the first user input to be a valid integer
		} catch (NumberFormatException firstInputException) {
			JOptionPane.showMessageDialog(null, "Please enter a valid integer for the number of crates.");
			continue;
		}
		
		while (!secondInputValid) {
	
			String averageNumberOfApples = JOptionPane.showInputDialog(null, "What is the average amount of apples that you expect inside of a crate?");
		
			if (averageNumberOfApples == null) {
				JOptionPane.showMessageDialog(null, "Cancelled.");
				// Terminates the execution and passes control back to enclosing loop for the dialog box
				continue;
			}
		
			// Using the try keyword to test for exception errors as the user attempts to pass in relevant input
			try {
				secondUserInput = Integer.parseInt(averageNumberOfApples);
				// Flip the initial input into a true boolean value
				secondInputValid = true;
				// This part handles the caught error within the try code block that is testing the first user input to be a valid integer
			} catch (NumberFormatException firstInputException) {
				JOptionPane.showMessageDialog(null, "Enter a valid integer for the average amount of whole apples expected per crate.");
				continue;
			}
			expectedAmount = firstUserInput * secondUserInput;
	    }
    }
    
    if (firstInputValid && secondInputValid == true) {
	// Print the expectedAmount of apples in the latest shipment
	    if (expectedAmount != -1) {
		    JOptionPane.showMessageDialog(null, "Expected number of apples in this shipment: " + expectedAmount);
	    }
    }
  }
}
```
